6-CREATE OR REPLACE FUNCTION verif_password (
    p_username IN VARCHAR2,
    p_password IN VARCHAR2
) RETURN BOOLEAN 
IS
BEGIN 
    IF (LENGTH(p_password) >= 6) AND 
       ((INSTR(p_password, '?') > 0) OR 
        (INSTR(p_password, '!') > 0) OR 
        (INSTR(p_password, '@') > 0)) AND 
       (p_username <> p_password) THEN
        RETURN TRUE;
    ELSE
        RETURN FALSE;
    END IF;
END;
